export const JSON = `{
  "status": "ok",
  "service": "staq-api",
  "env": "sandbox"
}`;

export const CREDENTIALS = `export CLIENT_ID="your-client-id"
export CLIENT_SECRET="your-client-secret"
export CERT_PATH="./certs/client.crt"
export KEY_PATH="./certs/client.key"`;

export const TOKEN_CURL = `ACCESS_TOKEN=$(
  curl -sS --fail \\
    --cert $CERT_PATH --key $KEY_PATH \\
    -u "$CLIENT_ID:$CLIENT_SECRET" \\
    -d "grant_type=client_credentials" \\
    "https://auth.staq.com/oauth/token" | jq -r '.access_token'
)`;

export const TEST_REQUEST_CURL = `{

    ## Postman set up

    Postman is an API client that lets you make calls without writing code. It’s the simplest way to test your connection and confirm your credentials are working.

    ### 1. Add your client certificate

    1. Go to Settings → Certificates
        - Host: sandbox.api.staq.io
        - CRT file: your signed certificate (client.crt)
        - KEY file: your private key (client.key)
    2. Click Add Certificate and enter the details.
    3. Save and ensure “SSL certificate verification” is turned on.

    ### 2. Create a new environment

    Add the following variables

    <div
      className="table"
      style={{ ['--table-cols']: '240px 1fr' }}
    >
      <div className="table--header">
        <span>Variable</span>
        <span>Example Value</span>
      </div>
      <div className="table--body">
        <div className="table--row">
          <span className="badge">BASE_URL</span>
          <span className="badge value">https://sandbox.api.staq.io</span>
        </div>

        <div className="table--row">
          <span className="badge">AUTH_URL</span>
          <span className="badge value">https://auth.staq.com/oauth/token</span>
        </div>

        <div className="table--row">
          <span className="badge">CLIENT_ID</span>
          <span className="value">Your app's client ID</span>
        </div>

        <div className="table--row">
          <span className="badge">CLIENT_SECRET</span>
          <span className="value">Your app’s Client Secret</span>
        </div>
      </div>
    </div>
    <div className="spacer"></div>
    ### 3. Generate an access token

    In Postman, open a new request and go to the Authorization tab.

    1. Select OAuth 2.0
    2. Click Get New Access Token
    3. Fill in the fields:
        - Token Name: Staq Sandbox
        - Grant Type: Client Credentials
        - Access Token URL: `{{AUTH_URL}}`
        - Client ID: `{{CLIENT_ID}}`
        - Client Secret: `{{CLIENT_SECRET}}`
        - Client Authentication: Send as Basic Auth header
    4. Click Get New Access Token. Postman will use your certificate to authenticate and return a token.
    5. Click Use Token — Postman will automatically add it to your next request.

    <div className="spacer"></div>

    #### 4. Make a test API call

    Create a new GET request to:

    <InlineCodeBlock code="{{BASE_URL}}/v1/ping" />

    Under the Authorization tab, ensure Bearer Token is selected and your token is filled in.

    Click Send.
  </Tab>
